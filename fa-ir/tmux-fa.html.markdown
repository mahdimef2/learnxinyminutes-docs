[tmux](http://tmux.github.io)
یک مالتی پلکسر ترمینال است که امکان ایجاد، دسترسی، و کنترل چندین ترمنال را دارد .
میتواند از یک صفحه ترمنال جدا شود در حالی که ترمینال به اجرای برنامه خود ادامه دهد و بعدا دوباره به ان متصل شود .


```

  tmux [command]     # اجرای دستورات
                     # تیماکس بدون دستور یک نشست جدید ایجاد میکند 

    new              # ایجاد نشست جدید
     -s "Session"    # ایجاد نشست جدید با نام 
     -n "Window"     # ایجاد پنجره جدید با نام
     -c "/dir"       # ایجاد نشست جدید در دایروکتوری مورد نظر

    attach           # اتصال به اخرین یا در دسترس بودن یک سشن 
     -t "#"          # اتصال به سشن مورد نظر
     -d              # قطع کردن سشن از نمونه های دیگر 

    ls               # لیست سشن های باز
     -a              # لیست تمام سشن های باز

    lsw              # لیست پنجره ها
     -a              # لیست تمام پنجره ها
     -s              # لیست تمام پنجره ها در این سشن

    lsp              # لیست پنل ها
     -a              # لیست همه پنل ها
     -s              # لیست همه پنل ها در این سشن
     -t              # لیست همه پنل ها در یک موقعیت خاص

    kill-window      # از پنجره فعلی خارج شو
     -t "#"          # پنجره مورد نظر را ببند
     -a              # همه پنجره‌ها را ببند
     -a -t "#"       # همه پنجره‌ها را ببند، به جز پنجره مورد نظر

    kill-session     # نشست فعلی را ببند
     -t "#"          # نشست مورد نظر را ببند
     -a              # همه نشست‌ها را ببند
     -a -t "#"       # همه نشست‌ها را ببند، به جز نشست مورد نظر

```


### دکمه های میانبر

روش کنترل یک جلسه متصل شده از طریق کلید های ترکیبی که به ان
کلید های "پیشوند" میگویند.  


```
----------------------------------------------------------------------
  (C-b) = Ctrl + b    # ترکیب 'پیشوند' مورد نیاز برای استفاده از دکمه‌های میانبر

  (M-1) = Meta + 1 -or- Alt + 1
----------------------------------------------------------------------

  ?                  # لیست تمام دکمه‌های میانبر
  :                  # وارد کردن حالت دستور tmux
  r                  # بازنشانی اجباری کلاینت متصل شده
  c                  # ایجاد یک پنجره جدید

  !                  # جدا کردن پنل فعلی از پنجره
  %                  # تقسیم پنل فعلی به دو قسمت چپ و راست
  "                  # تقسیم پنل فعلی به دو قسمت بالا و پایین

  n                  # تغییر به پنجره بعدی
  p                  # تغییر به پنجره قبلی
  {                  # جابجایی پنل فعلی با پنل قبلی
  }                  # جابجایی پنل فعلی با پنل بعدی
  [                  # ورود به حالت کپی برای کپی کردن متن یا مشاهده تاریخچه.

  s                  # انتخاب نشست جدید برای کلاینت متصل به صورت تعاملی
  w                  # انتخاب پنجره فعلی به صورت تعاملی
  0 to 9             # انتخاب پنجره از عدد ۰ تا ۹ 

  d                  # جدا کردن کلاینت فعلی
  D                  # انتخاب کلاینت برای جدا کردن

  &                  # بستن پنجره فعلی
  x                  # بستن پنل فعلی

  Up, Down           # تغییر به پنل بالایی، پایینی، چپ یا راست
  Left, Right

  M-1 تا M-5         # تنظیم پنل‌ها:
                       # 1) افقی-زوج
                       # 2) عمودی-زوج
                       # 3) افقی-اصلی
                       # 4) عمودی-اصلی
                       # 5) چیدمان تایل

  C-Up, C-Down       # تغییر اندازه پنل فعلی به صورت گام به گام
  C-Left, C-Right

  M-Up, M-Down       # تغییر اندازه پنل فعلی به صورت گام‌های پنج گانه
  M-Left, M-Right

  ‍‍‍```
